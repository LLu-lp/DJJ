## 1. 用户表 `users`

```sql
CREATE TABLE `users` (
  `id`            BIGINT UNSIGNED NOT NULL AUTO_INCREMENT COMMENT '主键ID',
  `username`      VARCHAR(32) NOT NULL COMMENT '用户名',
  `password`      VARCHAR(128) NOT NULL COMMENT '加密密码（bcrypt）',
  `phone`         VARCHAR(20) DEFAULT NULL COMMENT '手机号',
  `email`         VARCHAR(100) DEFAULT NULL COMMENT '邮箱',
  `nickname`      VARCHAR(50) DEFAULT NULL COMMENT '昵称',
  `avatar`        VARCHAR(255) DEFAULT NULL COMMENT '头像URL',
  `role`          TINYINT NOT NULL DEFAULT 0 COMMENT '角色：0-普通用户 1-员工 2-管理员',
  `status`        TINYINT NOT NULL DEFAULT 1 COMMENT '状态：0-禁用 1-正常',
  `last_login_at` DATETIME DEFAULT NULL COMMENT '最后登录时间',
  `created_at`    DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '创建时间',
  `updated_at`    DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP COMMENT '更新时间',
  PRIMARY KEY (`id`),
  UNIQUE KEY `uk_username` (`username`),
  UNIQUE KEY `uk_phone` (`phone`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='用户表';
```

| 字段 | 类型 | 说明 |
|------|------|------|
| id | BIGINT | 主键，自增 |
| username | VARCHAR(32) | 用户名，唯一 |
| password | VARCHAR(128) | bcrypt 加密密码 |
| phone | VARCHAR(20) | 手机号，唯一 |
| nickname | VARCHAR(50) | 昵称 |
| role | TINYINT | 0=用户 1=员工 2=管理员 |
| status | TINYINT | 0=禁用 1=正常 |

## 2. 电影表 `movies`

```sql
CREATE TABLE `movies` (
  `id`           BIGINT UNSIGNED NOT NULL AUTO_INCREMENT COMMENT '主键ID',
  `title`        VARCHAR(100) NOT NULL COMMENT '电影名称',
  `poster`       VARCHAR(255) DEFAULT NULL COMMENT '海报URL',
  `backdrop`     VARCHAR(255) DEFAULT NULL COMMENT '背景图URL',
  `genres`       VARCHAR(100) DEFAULT NULL COMMENT '类型，逗号分隔（动作,科幻）',
  `duration`     INT NOT NULL COMMENT '片长（分钟）',
  `director`     VARCHAR(100) DEFAULT NULL COMMENT '导演',
  `actors`       VARCHAR(500) DEFAULT NULL COMMENT '主演，逗号分隔',
  `synopsis`     TEXT DEFAULT NULL COMMENT '剧情简介',
  `language`     VARCHAR(50) DEFAULT '国语' COMMENT '语言版本',
  `rating`       DECIMAL(3,1) DEFAULT 0.0 COMMENT '评分（0-10）',
  `rating_count` INT DEFAULT 0 COMMENT '评分人数',
  `release_date` DATE DEFAULT NULL COMMENT '上映日期',
  `status`       TINYINT NOT NULL DEFAULT 0 COMMENT '状态：0-下架 1-即将上映 2-上映中',
  `is_hot`       TINYINT DEFAULT 0 COMMENT '是否热映：0-否 1-是',
  `ticket_count` INT DEFAULT 0 COMMENT '购票人数',
  `created_at`   DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
  `updated_at`   DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  PRIMARY KEY (`id`),
  KEY `idx_status` (`status`),
  KEY `idx_release_date` (`release_date`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='电影表';
```

## 6. 排片场次表 `schedules`

```sql
CREATE TABLE `schedules` (
  `id`          BIGINT UNSIGNED NOT NULL AUTO_INCREMENT,

  `cinema_id`   BIGINT UNSIGNED NOT NULL COMMENT '影院ID',
  `hall_id`     BIGINT UNSIGNED NOT NULL COMMENT '影厅ID',
  `show_date`   DATE NOT NULL COMMENT '放映日期',
  `show_time`   TIME NOT NULL COMMENT '开始时间',
  `end_time`    TIME NOT NULL COMMENT '结束时间',
  `price`       DECIMAL(8,2) NOT NULL COMMENT '普通票价',
  `vip_price`   DECIMAL(8,2) DEFAULT NULL COMMENT 'VIP票价',
  `language`    VARCHAR(20) DEFAULT '国语' COMMENT '语言版本',
  `sold_seats`  INT DEFAULT 0 COMMENT '已售座位数',
  `status`      TINYINT DEFAULT 1 COMMENT '状态：0-已取消 1-售票中 2-已结束',
  `created_at`  DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
  `updated_at`  DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  PRIMARY KEY (`id`),
  KEY `idx_movie_id` (`movie_id`),
  KEY `idx_cinema_date` (`cinema_id`, `show_date`),
  KEY `idx_show_date` (`show_date`),
  CONSTRAINT `fk_sch_movie` FOREIGN KEY (`movie_id`) REFERENCES `movies` (`id`),
  CONSTRAINT `fk_sch_cinema` FOREIGN KEY (`cinema_id`) REFERENCES `cinemas` (`id`),
  CONSTRAINT `fk_sch_hall` FOREIGN KEY (`hall_id`) REFERENCES `halls` (`id`)

```

## 8. 订单表 `orders`

```sql
CREATE TABLE `orders` (
  `id`           BIGINT UNSIGNED NOT NULL AUTO_INCREMENT,
  `order_no`     VARCHAR(30) NOT NULL COMMENT '订单号（JY+时间戳）',
  `user_id`      BIGINT UNSIGNED NOT NULL COMMENT '用户ID',
  `schedule_id`  BIGINT UNSIGNED NOT NULL COMMENT '场次ID',
  `movie_title`  VARCHAR(100) NOT NULL COMMENT '电影名称（冗余）',
  `show_time`    DATETIME NOT NULL COMMENT '放映时间（冗余）',
  `seat_count`   INT NOT NULL DEFAULT 1 COMMENT '购票张数',
  `ticket_amount` DECIMAL(10,2) NOT NULL COMMENT '票款金额',
  `snack_amount` DECIMAL(10,2) DEFAULT 0 COMMENT '小食金额',
  `discount_amount` DECIMAL(10,2) DEFAULT 0 COMMENT '优惠金额',
  `point_deduct` DECIMAL(10,2) DEFAULT 0 COMMENT '积分抵扣金额',
  `total_amount` DECIMAL(10,2) NOT NULL COMMENT '实付金额',
  `pay_type`     VARCHAR(20) DEFAULT NULL COMMENT '支付方式（wechat/alipay/memberCard）',
  `pay_time`     DATETIME DEFAULT NULL COMMENT '支付时间',
  `status`       TINYINT NOT NULL DEFAULT 0 COMMENT '状态：0-待支付 1-已支付 2-已退款 3-已取消 4-已核销',
  `ticketed`     TINYINT DEFAULT 0 COMMENT '是否已出票：0-否 1-是',
  `ticket_time`  DATETIME DEFAULT NULL COMMENT '出票时间',
  `qr_code`      VARCHAR(255) DEFAULT NULL COMMENT '取票二维码',
  `expire_time`  DATETIME DEFAULT NULL COMMENT '支付超时时间',
  `remark`       VARCHAR(500) DEFAULT NULL COMMENT '备注',
  `created_at`   DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
  `updated_at`   DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  PRIMARY KEY (`id`),
  UNIQUE KEY `uk_order_no` (`order_no`),
  KEY `idx_user_id` (`user_id`),
  KEY `idx_schedule_id` (`schedule_id`),
  KEY `idx_status` (`status`),
  KEY `idx_created_at` (`created_at`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='订单表';
```
