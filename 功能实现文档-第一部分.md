```java
@PostMapping("/login")
public Result<Map<String, Object>> login(@RequestBody LoginDTO dto) {
    // 1. 查询用户
    User user = userMapper.selectOne(
        new LambdaQueryWrapper<User>()
            .eq(User::getUsername, dto.getUsername())
            .eq(User::getDeleted, 0)
    );
    
    if (user == null) {
        return Result.fail("用户不存在");
    }
    
    // 2. 验证密码（使用 BCrypt）
    if (!passwordEncoder.matches(dto.getPassword(), user.getPassword())) {
        return Result.fail("密码错误");
    }
    
    // 3. 生成 JWT Token
    String token = jwtUtil.generateToken(user.getId(), user.getRole());
    
    // 4. 返回 Token 和用户信息
    return Result.success(Map.of(
        "token", token,
        "user", Map.of(
            "id", user.getId(),
            "username", user.getUsername(),
            "role", user.getRole()
        )

}
```

### 2.2 Token 验证

**JWT 工具类** (`backend/src/main/java/com/jiayue/cinema/util/JwtUtil.java`):
```java
public class JwtUtil {
    private String secret = "your-secret-key";
    private long expiration = 86400000; // 24 小时
    
    // 生成 Token
    public String generateToken(Long userId, Integer role) {
        return Jwts.builder()
            .setSubject(userId.toString())
            .claim("role", role)
            .setIssuedAt(new Date())
            .setExpiration(new Date(System.currentTimeMillis() + expiration))
            .signWith(SignatureAlgorithm.HS512, secret)
            .compact();
    }
    
    // 验证 Token
    public boolean validateToken(String token) {
        try {
            Jwts.parser().setSigningKey(secret).parseClaimsJws(token);
            return true;
        } catch (JwtException | IllegalArgumentException e) {
            return false;
        }
    }
    
    // 获取用户 ID
    public Long getUserIdFromToken(String token) {
        Claims claims = Jwts.parser()
            .setSigningKey(secret)
            .parseClaimsJws(token)
            .getBody();
        return Long.parseLong(claims.getSubject());
    }
}
```

**JWT 过滤器** (`backend/src/main/java/com/jiayue/cinema/security/JwtAuthenticationFilter.java`):
```java
@Component
public class JwtAuthenticationFilter extends OncePerRequestFilter {
    
    @Override
    protected void doFilterInternal(HttpServletRequest request, 
                                   HttpServletResponse response, 
                                   FilterChain filterChain) throws ServletException, IOException {
        // 1. 从请求头获取 Token
        String token = request.getHeader("Authorization");
        if (token != null && token.startsWith("Bearer ")) {
            token = token.substring(7);
            
            // 2. 验证 Token
            if (jwtUtil.validateToken(token)) {
                Long userId = jwtUtil.getUserIdFromToken(token);
                
                // 3. 设置认证信息
                SecurityContextHolder.getContext().setAuthentication(
                    new UsernamePasswordAuthenticationToken(userId, null, new ArrayList<>())
                );
            }
        }
        
        filterChain.doFilter(request, response);
    }
}
```

```java
@Transactional
public Order createOrder(CreateOrderDTO dto, Long userId) {
    // 1. 验证场次是否存在
    Schedule schedule = scheduleMapper.selectById(dto.getScheduleId());
    if (schedule == null) {
        throw new BusinessException("场次不存在");
    }
    
    // 2. 锁定座位（防止超卖）
    lockSeats(dto.getScheduleId(), dto.getSeatIds(), userId);
    
    // 3. 计算订单金额
    BigDecimal ticketAmount = calculateTicketAmount(schedule, dto.getSeatIds());
    BigDecimal snackAmount = calculateSnackAmount(dto.getSnackIds());
    BigDecimal totalAmount = ticketAmount.add(snackAmount);
    
    // 4. 应用会员折扣
    MemberInfo memberInfo = memberInfoMapper.selectOne(
        new LambdaQueryWrapper<MemberInfo>().eq(MemberInfo::getUserId, userId)
    );
    BigDecimal discountAmount = BigDecimal.ZERO;
    if (memberInfo != null && memberInfo.getLevelId() != null) {
        MemberLevel level = memberLevelMapper.selectById(memberInfo.getLevelId());
        discountAmount = ticketAmount.multiply(BigDecimal.ONE.subtract(level.getDiscount()));
    }
    
    // 5. 创建订单
    Order order = new Order();
    order.setOrderNo(generateOrderNo());
    order.setUserId(userId);
    order.setScheduleId(dto.getScheduleId());
    order.setTicketAmount(ticketAmount);
    order.setSnackAmount(snackAmount);
    order.setDiscountAmount(discountAmount);
    order.setPayAmount(totalAmount.subtract(discountAmount));
    order.setStatus(0);  // 待支付
    orderMapper.insert(order);
    
    // 6. 创建订单座位明细
    for (Long seatId : dto.getSeatIds()) {
        OrderItem item = new OrderItem();
        item.setOrderId(order.getId());
        item.setSeatId(seatId);
        orderItemMapper.insert(item);
    }
    
    return order;
}

// 锁定座位
private void lockSeats(Long scheduleId, List<Long> seatIds, Long userId) {
    for (Long seatId : seatIds) {
        ScheduleSeat scheduleSeat = scheduleSeatMapper.selectOne(
            new LambdaQueryWrapper<ScheduleSeat>()
                .eq(ScheduleSeat::getScheduleId, scheduleId)
                .eq(ScheduleSeat::getSeatId, seatId)
        );
        
        if (scheduleSeat == null || scheduleSeat.getStatus() != 0) {
            throw new BusinessException("座位不可用");
        }
        
        // 更新座位状态为锁定（2）
        scheduleSeatMapper.update(null,
            new LambdaUpdateWrapper<ScheduleSeat>()
                .eq(ScheduleSeat::getId, scheduleSeat.getId())
                .set(ScheduleSeat::getStatus, 2)
                .set(ScheduleSeat::getLockedBy, userId)
                .set(ScheduleSeat::getLockedAt, LocalDateTime.now())
        );
    }
}
```

### 3.4 座位状态说明

|------|------|------|
| 0 | 可选 | 座位可以被选购 |
| 1 | 已售 | 座位已被购买 |
| 2 | 锁定 | 座位被用户锁定，等待支付（5分钟后自动解锁） |

### 4.2 会员等级自动升级

**核心逻辑** (`backend/src/main/java/com/jiayue/cinema/service/impl/MemberServiceImpl.java`):

```java
// 在获取会员信息时检查并升级等级
@Override
public MemberInfoVO getMemberInfo(Long userId) {
    MemberInfo memberInfo = memberInfoMapper.selectOne(
        new LambdaQueryWrapper<MemberInfo>().eq(MemberInfo::getUserId, userId)
    );
    
    if (memberInfo != null) {
        // 根据累计消费金额重新检查并升级会员等级
        checkAndUpgradeLevel(userId, memberInfo.getTotalSpent());
        
        // 重新获取最新的会员信息
        memberInfo = memberInfoMapper.selectOne(
            new LambdaQueryWrapper<MemberInfo>().eq(MemberInfo::getUserId, userId)
        );
    }
    
    // ... 返回会员信息
}

// 检查并升级等级
private void checkAndUpgradeLevel(Long userId, BigDecimal totalSpent) {
    // 查询满足条件的最高等级
    List<MemberLevel> levels = memberLevelMapper.selectList(
        new LambdaQueryWrapper<MemberLevel>()
            .le(MemberLevel::getMinSpent, totalSpent)  // 消费额 >= 最低消费额
            .orderByDesc(MemberLevel::getMinSpent)     // 按消费额降序
            .last("LIMIT 1")
    );
    
    if (!levels.isEmpty()) {
        MemberLevel level = levels.get(0);
        // 更新用户的会员等级
        memberInfoMapper.update(null,
            new LambdaUpdateWrapper<MemberInfo>()
                .eq(MemberInfo::getUserId, userId)
                .set(MemberInfo::getLevelId, level.getId())
        );
    }
}

// 更新累计消费金额
@Transactional
public void updateTotalSpent(Long userId, BigDecimal amount) {
    MemberInfo memberInfo = memberInfoMapper.selectOne(
        new LambdaQueryWrapper<MemberInfo>().eq(MemberInfo::getUserId, userId)
    );
    
    if (memberInfo != null) {
        BigDecimal newTotalSpent = memberInfo.getTotalSpent().add(amount);
        memberInfo.setTotalSpent(newTotalSpent);
        memberInfoMapper.updateById(memberInfo);
        
        // 检查并升级等级
        checkAndUpgradeLevel(userId, newTotalSpent);
    }
}
```

```java
@PostMapping("/items/{itemId}")
@Transactional
public Result<Void> exchange(@PathVariable Long itemId) {
    Long userId = authService.getCurrentUserId();
    
    // 1. 获取兑换商品
    ExchangeItem item = exchangeItemMapper.selectById(itemId);
    if (item == null || item.getStatus() == 0) {
        return Result.fail("商品不存在或已下架");
    }
    
    // 2. 检查用户积分
    MemberInfo memberInfo = memberInfoMapper.selectOne(
        new LambdaQueryWrapper<MemberInfo>().eq(MemberInfo::getUserId, userId)
    );
    
    if (memberInfo.getPoints() < item.getPoints()) {
        return Result.fail("积分不足");
    }
    
    // 3. 扣除积分
    memberInfo.setPoints(memberInfo.getPoints() - item.getPoints());
    memberInfoMapper.updateById(memberInfo);
    
    // 4. 创建兑换记录
    ExchangeRecord record = new ExchangeRecord();
    record.setUserId(userId);
    record.setItemId(itemId);
    record.setItemName(item.getName());
    record.setPoints(item.getPoints());
    record.setStatus(0);  // 待领取
    exchangeRecordMapper.insert(record);
    
    return Result.success();
}
```

```java
public Result<Map<String, Object>> verifyTicket(@RequestBody Map<String, String> request) {
    String ticketCode = request.get("code");
    
    // 1. 查询订单
    Order order = orderMapper.selectOne(
        new LambdaQueryWrapper<Order>()
            .eq(Order::getStatus, 1)  // 已支付

                .eq(Order::getQrCode, ticketCode)

                .eq(Order::getOrderNo, ticketCode))
```

```java
private List<ScheduleVO> convertToVOList(List<Schedule> schedules) {
    // ... 其他代码 ...

        ScheduleVO vo = new ScheduleVO();

            new LambdaQueryWrapper<ScheduleSeat>()

                .eq(ScheduleSeat::getStatus, 0)

        // 计算已售座位数

            new LambdaQueryWrapper<ScheduleSeat>()

        vo.setAvailableSeats(availableSeats.intValue());
        vo.setSoldSeats(soldSeats.intValue());
        
        return vo;
    }).toList();
}
```
