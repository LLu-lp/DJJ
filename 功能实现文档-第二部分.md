public Result<Void> addMemberLevel(@RequestBody MemberLevel memberLevel) {
    memberLevelMapper.insert(memberLevel);

@PutMapping("/member-levels/{level}")
public Result<Void> updateMemberLevel(
        @PathVariable Integer level,
        @RequestBody MemberLevel memberLevel) {
    memberLevel.setLevel(level);
    memberLevelMapper.update(memberLevel,
        new LambdaQueryWrapper<MemberLevel>().eq(MemberLevel::getLevel, level)

    return Result.success();

@DeleteMapping("/member-levels/{level}")
public Result<Void> deleteMemberLevel(@PathVariable Integer level) {
    memberLevelMapper.delete(
        new LambdaQueryWrapper<MemberLevel>().eq(MemberLevel::getLevel, level)
    );
    return Result.success();
}

@GetMapping("/orders")
public Result<Page<Order>> getOrders(
        @RequestParam(defaultValue = "1") Integer page,
        @RequestParam(defaultValue = "10") Integer size,
        @RequestParam(required = false) Integer status,
        @RequestParam(required = false) String keyword) {
    
    LambdaQueryWrapper<Order> wrapper = new LambdaQueryWrapper<>();
    wrapper.eq(status != null, Order::getStatus, status)
           .and(keyword != null && !keyword.isBlank(), w -> w
               .like(Order::getOrderNo, keyword)
               .or()
               .like(Order::getUsername, keyword))
           .orderByDesc(Order::getCreatedAt);
    
    return Result.success(orderMapper.selectPage(new Page<>(page, size), wrapper));
}

// 订单退款
@PostMapping("/orders/{id}/refund")
@Transactional
public Result<Void> refundOrder(@PathVariable Long id) {
    Order order = orderMapper.selectById(id);
    if (order == null) {
        return Result.fail("订单不存在");
    }
    
    if (order.getStatus() != 1) {
        return Result.fail("只能退款已支付的订单");
    }
    
    // 1. 更新订单状态为已退款
    order.setStatus(3);
    orderMapper.updateById(order);
    
    // 2. 解锁座位
    List<OrderItem> items = orderItemMapper.selectList(
        new LambdaQueryWrapper<OrderItem>().eq(OrderItem::getOrderId, id)
    );
    
    for (OrderItem item : items) {
        scheduleSeatMapper.update(null,
            new LambdaUpdateWrapper<ScheduleSeat>()
                .eq(ScheduleSeat::getScheduleId, order.getScheduleId())
                .eq(ScheduleSeat::getSeatId, item.getSeatId())
                .set(ScheduleSeat::getStatus, 0)  // 恢复为可选
        );
    }
    
    // 3. 返还金额到用户账户
    User user = userMapper.selectById(order.getUserId());
    user.setBalance(user.getBalance().add(order.getPayAmount()));
    userMapper.updateById(user);
    
    return Result.success();
}

```java
@PostMapping("/{orderId}/pay")
@Transactional
public Result<Void> payOrder(
        @PathVariable Long orderId,
        @RequestParam Integer payType) {
    
    Long userId = authService.getCurrentUserId();
    Order order = orderMapper.selectById(orderId);
    
    if (order == null || !order.getUserId().equals(userId)) {
        return Result.fail("订单不存在");
    }
    
    if (order.getStatus() != 0) {
        return Result.fail("订单状态不正确");
    }
    
    // 1. 更新订单状态为已支付
    order.setStatus(1);
    order.setPayType(payType);
    order.setPayTime(LocalDateTime.now());
    orderMapper.updateById(order);
    
    // 2. 更新座位状态为已售
    List<OrderItem> items = orderItemMapper.selectList(
        new LambdaQueryWrapper<OrderItem>().eq(OrderItem::getOrderId, orderId)
    );
    
    for (OrderItem item : items) {
        scheduleSeatMapper.update(null,
            new LambdaUpdateWrapper<ScheduleSeat>()
                .eq(ScheduleSeat::getScheduleId, order.getScheduleId())
                .eq(ScheduleSeat::getSeatId, item.getSeatId())
                .set(ScheduleSeat::getStatus, 1)  // 已售
        );
    }
    
    // 3. 更新会员消费金额和积分
    memberService.updateTotalSpent(userId, order.getTicketAmount());
    
    // 4. 获取会员信息计算积分
    MemberInfo memberInfo = memberInfoMapper.selectOne(
        new LambdaQueryWrapper<MemberInfo>().eq(MemberInfo::getUserId, userId)
    );
    
    if (memberInfo != null) {
        MemberLevel level = memberLevelMapper.selectById(memberInfo.getLevelId());
        if (level != null) {
            // 计算获得的积分
            int points = order.getTicketAmount().multiply(level.getPointsRate()).intValue();
            memberService.addPoints(userId, points, 1, orderId, "购票获得积分");
        }
    }
    
    return Result.success();
}
```

## 十三、常见问题解决

### 13.1 座位超卖问题

**问题**: 多个用户同时选座可能导致座位超卖

**解决方案**: 使用座位锁定机制
- 用户选座时，座位状态变为 2（锁定）
- 5 分钟内未支付，座位自动解锁
- 支付成功后，座位状态变为 1（已售）

**实现代码**:
```java
// 锁定座位
scheduleSeatMapper.update(null,
    new LambdaUpdateWrapper<ScheduleSeat>()
        .eq(ScheduleSeat::getId, scheduleSeat.getId())
        .set(ScheduleSeat::getStatus, 2)
        .set(ScheduleSeat::getLockedBy, userId)
        .set(ScheduleSeat::getLockedAt, LocalDateTime.now())
);

// 定时任务：5分钟后自动解锁
@Scheduled(fixedDelay = 300000)  // 5分钟
public void unlockExpiredSeats() {
    LocalDateTime fiveMinutesAgo = LocalDateTime.now().minusMinutes(5);
    scheduleSeatMapper.update(null,
        new LambdaUpdateWrapper<ScheduleSeat>()
            .eq(ScheduleSeat::getStatus, 2)
            .lt(ScheduleSeat::getLockedAt, fiveMinutesAgo)
            .set(ScheduleSeat::getStatus, 0)
            .set(ScheduleSeat::getLockedBy, null)
            .set(ScheduleSeat::getLockedAt, null)
    );
}
```

### 13.2 会员等级升级不生效

**问题**: 用户消费达到升级条件但等级未更新

**原因**: 只在 `updateTotalSpent` 时检查升级，获取会员信息时没有重新检查

**解决方案**: 在 `getMemberInfo` 中添加升级检查
```java
@Override
public MemberInfoVO getMemberInfo(Long userId) {
    MemberInfo memberInfo = memberInfoMapper.selectOne(
        new LambdaQueryWrapper<MemberInfo>().eq(MemberInfo::getUserId, userId)
    );
    
    if (memberInfo != null) {
        // 根据累计消费金额重新检查并升级会员等级
        checkAndUpgradeLevel(userId, memberInfo.getTotalSpent());
        
        // 重新获取最新的会员信息
        memberInfo = memberInfoMapper.selectOne(
            new LambdaQueryWrapper<MemberInfo>().eq(MemberInfo::getUserId, userId)
        );
    }
    
    // ... 返回会员信息
}
```

### 13.3 已售座位显示为 0

**问题**: 员工端显示的已售座位数一直是 0

**原因**: 后端只计算了可用座位数，没有计算已售座位数

**解决方案**: 在 `ScheduleVO` 中添加 `soldSeats` 字段并计算
```java
// 计算已售座位数
Long soldSeats = scheduleSeatMapper.selectCount(
    new LambdaQueryWrapper<ScheduleSeat>()
        .eq(ScheduleSeat::getScheduleId, schedule.getId())
        .eq(ScheduleSeat::getStatus, 1)
);
vo.setSoldSeats(soldSeats.intValue());
```

## 十四、性能优化建议

### 14.1 数据库优化
- 为 `orders.user_id`、`orders.schedule_id` 添加索引
- 为 `schedule_seats.schedule_id`、`schedule_seats.status` 添加联合索引
- 为 `member_info.user_id` 添加唯一索引

### 14.2 缓存优化
- 使用 Redis 缓存热门电影信息
- 缓存会员等级配置（变化不频繁）
- 缓存场次座位状态（实时性要求高，谨慎使用）

### 14.3 查询优化
- 使用分页查询大数据集
- 避免 N+1 查询问题，使用批量查询
- 使用 MyBatis Plus 的 `selectBatchIds` 方法

## 十五、安全建议

### 15.1 认证与授权
- 使用 JWT Token，设置合理的过期时间
- 实现权限拦截器，验证用户角色
- 敏感操作添加二次验证

### 15.2 数据安全
- 密码使用 BCrypt 加密存储
- 敏感信息（如支付信息）不在日志中输出
- 实现操作审计日志

### 15.3 API 安全
- 实现 API 速率限制，防止暴力攻击
- 验证所有用户输入，防止 SQL 注入
- 使用 HTTPS 加密传输
